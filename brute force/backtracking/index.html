Quay lui

Giới thiệu:

  • Thuật toán Quay lui (backtracking) dùng để giải bài toán liệt kê các cấu hình. Mỗi cấu hình được xây dựng bằng cách xây dựng từng phần tử, mỗi phần tử được chọn bằng cách thử tất cả các khả năng.

Ví dụ của Quay lui trong bài toán duyệt nhị phân:

#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector<int> a(n + 1, 0);
    function<void(int)> backtrack = [&](int i) {
        if (i == n + 1) {
            for (int j = 1; j <= n; ++j) {
                cout << a[j] << " ";
            }
            cout << endl;
            return;
        }
        for (int j = 0; j <= 1; ++j) {
            a[i] = j;
            backtrack(i + 1);
        }
    };
    backtrack(1);
    return 0;
}
