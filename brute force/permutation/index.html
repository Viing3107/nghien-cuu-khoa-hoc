Hoán vị

Giới thiệu:

  • Một hoán vị là sự sắp xếp lại tất cả các phần tử của một chuỗi.

Ví dụ của Hoán vị trong bài toán liệt kê dãy hoán vị:

#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector<int> a(n + 1, 0);
    vector<bool> used(n + 1, false);
    function<void(int)> backtrack = [&](int i) {
        if (i == n + 1) {
            for (int j = 1; j <= n; ++j) {
                cout << a[j] << " ";
            }
            cout << endl;
            return;
        }
        for (int j = 1; j <= n; ++j) {
            if (!used[j]) {
                a[i] = j;
                used[j] = true;
                backtrack(i + 1);
                used[j] = false;
            }
        }
    };
    backtrack(1);
    return 0;
}
